<Statements> ::=
  "" | <Statement> | <Statement> <Statements> ;

<Statement> ::=
  <AssignmentStatement> |
  <DoWhileStatement> |
  <UndefinedStatement> ;

<AssignmentStatement> ::=
  <Ident> "Equals" <Expression> ;

<DoWhileStatement> ::=
  "Do" <Statements> "While" <Expression> ;

<UndefinedStatement> ::=
  "" ;

<Expression> ::=
  <BinaryExpression> |
  <PrimaryExpression> |
  <EmptyExpression> ;

<PrimaryExpression> ::=
  <IdentExpression> |
  <LiteralExpression> ;

<LiteralExpression> ::=
  <NumberLiteral> |
  <StringLiteral> ;

<BinaryExpression> ::=
  <Expression> <Op> <Expression> ;

<Op> ::=
  "Add" | "Minus" | "Divide" | "Multiply" ;
